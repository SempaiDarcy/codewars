function yearDays(year)
{
  let days = '';
  if((year % 4 == 0) && (year % 100 != 0) || (year % 400 == 0)){
    days = year + ' has 366 days';
  }
  else if(year%100===0 && year%400 !== 0){
    days = year + ' has 365 days';
  }
  else{
    days = year + ' has 365 days';
  }
  return days;
}
